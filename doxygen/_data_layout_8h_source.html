458  template <class T>
459  T* getVarData(size_t varPieceIndex, size_t& outCount) {
460  if (mappedDataLayout_ != nullptr) {
461  return mappedDataLayout_->getVarData<T>(varPieceIndex, outCount);
462  }
463  if (varPieceIndex < varSizePieces_.size()) {
464  const IndexEntry& indexEntry = getVarSizeIndex()[varPieceIndex];
465  if (indexEntry.getOffset() + indexEntry.getLength() <= varData_.size()) {
466  outCount = indexEntry.getLength() / sizeof(T);
467  return reinterpret_cast<T*>(varData_.data() + indexEntry.getOffset());
468  }
469  }
470  outCount = 0;
471  return nullptr;
472  }
